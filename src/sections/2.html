¶ Структура проекта и инстурменты

Для разработки игры мы будем использовать:

  • Mocha для запуска тестов;
  • Chai для проверки условий;
  • Sinon для проверки запуска функций и методов;
  • JSDOM для проверки функций, работающих с ДОМ-деревом;
  • Babel для работы с новым синтаксисом языка;
  • Webpack для сборки проекта.
// package.json
"scripts": {
  "test": "mocha --require babel-core/register",
  "build": "webpack"
},

Создадим конфигурацию сборки и пропишем входной файл. Сборка нам понадобится ближе к концу разработки, но чтобы не отвлекаться в будущем, настроим её сейчас.

// webpack.config.js
var webpack = require('webpack')
var path = require('path')

module.exports = [{
  name: 'client',
  mode: 'production',
  context: path.resolve(__dirname),
  entry: {
    javascript: ['babel-polyfill', './src/index.js']
  },
  output: {
    filename: './js/bundle.js',
    path: path.resolve(__dirname, 'dist'),
  },
}]
